import { CardGrid, LinkCard } from '@astrojs/starlight/components'
import { getCollection } from 'astro:content'
import { LINKS } from '~/constants/links'

export const projectGuides = (await getCollection('docs'))
	.filter(entry => entry.id.startsWith('learn/guides/project/') && entry.id !== 'learn/guides/project/index.mdx')
	.sort((a, b) => a.data.title.localeCompare(b.data.title))

<CardGrid>
	{projectGuides.map(guide => (
		<LinkCard 
			title={guide.data.title} 
			href={LINKS.guides(guide.id.replace('learn/guides/', ''))} 
		/>
	))}
</CardGrid>
