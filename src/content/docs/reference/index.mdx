import { CardGrid, LinkCard } from '@astrojs/starlight/components'
import { getCollection } from 'astro:content'

export const referencePages = (await getCollection('docs'))
	.filter(entry => entry.id.startsWith('reference/') && entry.id !== 'reference/index.mdx')
	.sort((a, b) => a.id.localeCompare(b.id))

<CardGrid>
	{referencePages.map(page => (
		<LinkCard 
			title={page.data.title} 
			href={`/my-learning-curve/reference/${page.id.replace('reference/', '').replace('.md', '').replace('.mdx', '')}`} 
		/>
	))}
</CardGrid>
